Momentum Plugin for Y.DD

Adds momentum to elements that are draggable. Supports plain 'ol dd, dd-plugin, and dd-delegate.

The direction and distance that an object will travel is decided by sampling the direction and speed of the mouse during a drag. When the user lets go of the left mouse button the dragged DOM node is moved via a lightweight JS animation in the terminal direction until the applied drag reduces the momentum to a point where it makes sense to end the animation.

Tested against IE6/7/8/9/10, FF, Chrome, Safari.

Example Code

// Base case 1 - simple dd plugin, verbose syntax
YUI().use("dd-drag", "plugin", "gallery-dd-plugin-momentum", function(Y){
  // assuming #demo exists on the page
  var dd = new Y.DD.Drag({
    node: '#demo'
  });

  // add momentum to the drag
  dd.plug(Y.DDMomentumPlugin);

});

// Base case 2 - simple dd plugin, chaining syntax
YUI().use("dd-drag", "plugin", "gallery-dd-plugin-momentum", function(Y){
  // assuming #demo exists on the page
  var dd = new Y.DD.Drag({
    node: '#demo'
  }).plug(Y.DDMomentumPlugin);

});

// Delegate dragging
YUI().use("dd-delegate", "plugin", "gallery-dd-plugin-momentum", function(Y) {
  // assuming a node with id=demo that contains 'li' nodes exists on the page
  var del = new Y.DD.Delegate({
      container: '#demo',
      nodes: 'li'
  }).plug(Y.DDMomentumPlugin);

});

// Plugin dragging
YUI().use("dd-plugin", "plugin", "gallery-dd-plugin-momentum", function(Y){
  // assuming #demo exists on the page
  var node = Y.one('#demo');

  // add drag to the node
  node.plug(Y.Plugin.Drag);

  // add momentum to the drag
  node.dd.plug(Y.DDMomentumPlugin);
});
